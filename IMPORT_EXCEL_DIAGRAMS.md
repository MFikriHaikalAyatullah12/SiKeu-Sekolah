## Transaction Type Detection Algorithm

```javascript
function determineTransactionType(accountName, coaAccount) {
  // Priority 1: COA Type
  if (coaAccount?.subCategory?.category?.type === 'REVENUE') 
    return 'INCOME'
  
  if (coaAccount?.subCategory?.category?.type === 'EXPENSE')
    return 'EXPENSE'
  
  // Priority 2: Keyword Matching
  const lowerName = accountName.toLowerCase()
  
  const incomeKeywords = [
    'pendapatan', 'pemasukan', 'penerimaan',
    'income', 'revenue', 'spp', 'uang sekolah',
    'donasi', 'bantuan'
  ]
  
  const expenseKeywords = [
    'pengeluaran', 'biaya', 'belanja',
    'expense', 'beban', 'operasional',
    'gaji', 'honor'
  ]
  
  if (incomeKeywords.some(kw => lowerName.includes(kw)))
    return 'INCOME'
  
  if (expenseKeywords.some(kw => lowerName.includes(kw)))
    return 'EXPENSE'
  
  // Default
  return 'EXPENSE'
}
```

- PDF generation failure tidak menggagalkan transaksi
